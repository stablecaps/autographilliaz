function check_alias_clashes() {

    alias | sed 's/^[^ ]* *\|=.*$//g' | while read a; do
    printf "%20.20s : %s\n" $a "$(type -ta $a | tr '\n' ' ')"
    done | awk -F: '$2 ~ /file/'
}

function up() {

    local d=""
    limit=$1
    for ((i=1 ; i <= limit ; i++))
        do
            d=$d/..
        done
    d=$(echo $d | sed 's/^\///')
    if [ -z "$d" ]; then
        d=..
    fi
    cd $d
}







help() {
    "$@" --help 2>&1 | bathelp
}
